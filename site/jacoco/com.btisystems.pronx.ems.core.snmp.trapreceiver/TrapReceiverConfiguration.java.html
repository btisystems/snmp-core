package com.btisystems.pronx.ems.core.snmp.trapreceiver;

public class TrapReceiverConfiguration implements ITrapReceiverConfiguration {
    private ITrapHandlerService trapHandlerService;
    private ITrapSourceMapper addressMapper;
    private int dispatcherThreadCount;
    private String listeningAddress;

    public void setAddressMapper(final ITrapSourceMapper addressMapper) {
        this.addressMapper = addressMapper;
    }

    public void setDispatcherThreadCount(final int dispatcherThreadCount) {
        this.dispatcherThreadCount = dispatcherThreadCount;
    }

    public void setListeningAddress(final String listeningAddress) {
        this.listeningAddress = listeningAddress;
    }

    public void setTrapHandlerService(final ITrapHandlerService trapHandlerService) {
        this.trapHandlerService = trapHandlerService;
    }
    
    @Override
    public ITrapHandlerService getTrapHandlerService() {
        return trapHandlerService;
    }

    @Override
    public ITrapSourceMapper getAddressMapper() {
        return addressMapper;
    }

    @Override
    public int getDispatcherThreadCount() {
        return dispatcherThreadCount;
    }

    @Override
    public String getListeningAddress() {
        return listeningAddress;
    }
    
}
