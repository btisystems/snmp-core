package com.btisystems.pronx.ems.core.snmp;


public class ReadWriteSnmpConfigurationFactory implements ISnmpConfigurationFactory {
    private final ISnmpConfiguration readConfig;
    private final ISnmpConfiguration writeConfig;

    public ReadWriteSnmpConfigurationFactory(final ISnmpConfiguration readConfig, final ISnmpConfiguration writeConfig) {
        this.readConfig = readConfig;
        this.writeConfig = writeConfig;
    }
    
    @Override
    public ISnmpConfiguration getConfiguration(final AccessType accessType) {
        if (accessType == AccessType.READ_ONLY){
            return readConfig;
        } else {
            return writeConfig;
        }
    }
    
}
